				}
			}
		}
		hidden_effect = {
			save_scope_value_as = {
				name = restart_scheme
				value = yes
			}
		}
	}

	after = {
		hidden_effect = {
			if = {
				limit = {
					exists = scope:restart_scheme
					exists = scope:scheme
				}

				}
			}
			else_if = {

					exists = scope:scheme
					exists = scope:target
					scope:target = { is_alive = yes }
				}
				scope:scheme = {
					end_scheme = yes
				}
			}
		}
	}
}
