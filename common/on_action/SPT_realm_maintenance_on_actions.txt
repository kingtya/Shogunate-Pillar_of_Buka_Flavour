realm_maintenance_quarterly_pulse = {
	on_actions = {
		spt_policy_canal_improvements_pulse
		spt_policy_laws_of_imagawa_pulse
		spt_policy_laws_of_takeda_pulse 
		spt_policy_five_admonitions_pulse
	}
}

spt_policy_canal_improvements_pulse = {
	trigger = {
		has_character_modifier = spt_policy_canal_improvements_modifier
		any_held_county = {
			NOT = { has_county_modifier = spt_policy_canal_improvements_county_modifier }
		}
	}
	effect = {
		every_held_county = {
			limit = {
				NOT = { has_county_modifier = spt_policy_canal_improvements_county_modifier }
			}
			random = {
				chance = 50
				save_scope_as = county
				holder = {
					save_scope_as = county_holder
					trigger_event = {
						id = spt_policy_events.0003
						days = { 10 40 }
					}
				}
			}
		}
	}
}
