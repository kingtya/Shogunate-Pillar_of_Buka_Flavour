SPT_kkt_init_flag_list = {
	scope = character

	effect = {
		SPT_init_neutral_poilcy_flag_list = yes
		SPT_init_conservative_poilcy_flag_list = yes
		SPT_init_progressive_poilcy_flag_list = yes
	}
}

SPT_kkt_exit_flag_list = {
	scope = character

	effect = {
		SPT_clear_flag_list = yes
	}
}


SPT_kkt_remove_all_kkt_modifier = {
	scope = character

	effect = {
		remove_character_modifier = flag_kkt_1_modifier
		remove_character_modifier = flag_kkt_2_modifier
		remove_character_modifier = flag_kkt_3_modifier
		remove_character_modifier = flag_kkt_4_modifier
		remove_character_modifier = flag_kkt_5_modifier
		remove_character_modifier = flag_kkt_6_modifier
	}
}

SPT_kkt_apply_modifier = {
	scope = character

	saved_scopes = {
		SPT_kkt_selected_flag
	}

	effect = {
		SPT_kkt_set_modifier_to_character = {
			FLAG = scope:SPT_kkt_selected_flag
			ACTION = add
			CONFIRM = yes
		}
	}
}

SPT_kkt_can_apply_modifier = {
	scope = character

	saved_scopes = {
		SPT_kkt_selected_flag
	}

	is_valid = {
		NOT = {
			SPT_kkt_is_policy_released = {
				FLAG = scope:SPT_kkt_selected_flag
			}
		}
	}
}

SPT_kkt_remove_modifier = {
	scope = character

	saved_scopes = {
		SPT_kkt_selected_flag
	}

	effect = {
		SPT_kkt_set_modifier_to_character = {
			FLAG = scope:SPT_kkt_selected_flag
			ACTION = remove
			CONFIRM = no
		}
	}
}

SPT_kkt_can_remove_modifier = {
	scope = character

	saved_scopes = {
		SPT_kkt_selected_flag
	}

	is_valid = {
		SPT_kkt_is_policy_released = {
			FLAG = scope:SPT_kkt_selected_flag
		}
	}
}


SPT_kkt_apply_cost = {
	scope = character

	saved_scopes = {
		SPT_kkt_selected_flag
	}

	effect = {
		SPT_kkt_set_cost_to_character = {
			FLAG = scope:SPT_kkt_selected_flag
		}
	}
}
