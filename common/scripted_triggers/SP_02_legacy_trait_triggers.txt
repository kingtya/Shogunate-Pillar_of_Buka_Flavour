sp_has_imagawa_margin_trait = {
	OR = {
		has_trait = imagawa_margin
		AND = {
			exists = cp:councillor_spt_bugyo
			cp:councillor_spt_bugyo = {
				has_trait = imagawa_margin
				stewardship >= decent_skill_rating
			}
		}
	}
}

	NOT = { has_trait = imagawa_margin }
	exists = cp:councillor_spt_bugyo
	cp:councillor_spt_bugyo = {
		has_trait = imagawa_margin
	}
}
