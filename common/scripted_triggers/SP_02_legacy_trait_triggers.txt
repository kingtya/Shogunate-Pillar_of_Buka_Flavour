sp_has_legacy_trait_trigger = {
	OR = {
		has_trait = $TRAIT$
		AND = {
			exists = cp:councillor_spt_bugyo
			cp:councillor_spt_bugyo = {
				has_trait = $TRAIT$
				$SKILL$ >= $RATING$
			}
		}
	}
}
sp_councillor_has_legacy_trait_trigger = {
	NOT = { has_trait = $TRAIT$ }
	exists = cp:councillor_spt_bugyo
	cp:councillor_spt_bugyo = {
		has_trait = $TRAIT$
	}
}
