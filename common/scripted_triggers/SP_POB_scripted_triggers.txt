## 撤销政策
sp_can_revoke_policy = {
	trigger_if = {
		limit = {
			exists = $FLAG$
		}
		custom_description = {
			text = sp_policy_not_released_cd
			trigger_if = {
				limit = {
					has_variable_list = sp_released_policy_flags
				}
				any_in_list = {
					variable = sp_released_policy_flags
					this = $FLAG$
				}
			}
			trigger_else = {
				always = no
			}
		}

		custom_description = {
			text = sp_policy_released_recently_cd
			trigger_if = {
				limit = {
					trigger_if = {
						limit = {
							has_variable_list = sp_released_policy_flags
						}
						any_in_list = {
							variable = sp_released_policy_flags
							this = $FLAG$
						}
						
					}
					trigger_else = {
						always = no
					}
				}
				NOT = {
					has_character_flag = $FLAG$
				}
			}
			trigger_else = {
				always = yes
			}
		}
	}
	trigger_else = {
		always = no
	}
	
}
## 已发布政策
sp_is_released_policy = {
	trigger_if = {
		limit = {
			has_variable_list = sp_released_policy_flags
		}
		any_in_list = {
			variable = sp_released_policy_flags
			this = $FLAG$
		}
	}
	trigger_else = {
		always = no
	}
}
