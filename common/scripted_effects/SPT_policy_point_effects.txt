SPT_init_points = {
	if = {
		limit = {
			NOT = {
				exists = var:current_kkt_points
			}
		}
		set_variable = {
			name = current_kkt_points
			value = 0
		}
	}
}

SPT_kkt_count_pocliy_points = {
	set_variable = {
		name = SPT_will_be_kkt_points
		value = {
			value = root.var:current_kkt_points
			add = $FLAG$.SPT_kkt_points
		}
	}
}
