SPT_init_flag_list = {
	# 注：此处每一个flag对应一个国策，root为当前角色，如果需要动态显示国策，用if判断即可
	add_to_variable_list = {
		name = SPT_kkt_flags
		target = flag:flag_kkt_1
	}
	add_to_variable_list = {
		name = SPT_kkt_flags
		target = flag:flag_kkt_2
	}
	add_to_variable_list = {
		name = SPT_kkt_flags
		target = flag:flag_kkt_3
	}
	add_to_variable_list = {
		name = SPT_kkt_flags
		target = flag:flag_kkt_4
	}
	add_to_variable_list = {
		name = SPT_kkt_flags
		target = flag:flag_kkt_5
	}
	add_to_variable_list = {
		name = SPT_kkt_flags
		target = flag:flag_kkt_6
	}
}

SPT_clear_flag_list = {
	clear_variable_list = SPT_kkt_flags
}

SPT_kkt_set_modifier_to_character = {
	show_as_tooltip = { # 作为显示效果，不实际执行
		switch = {
			trigger = $FLAG$
	
			flag:flag_kkt_1 = {
				$ACTION$_character_modifier = flag_kkt_1_modifier
			}
			flag:flag_kkt_2 = {
				$ACTION$_character_modifier = flag_kkt_2_modifier
			}
			flag:flag_kkt_3 = {
				$ACTION$_character_modifier = flag_kkt_3_modifier
			}
			flag:flag_kkt_4 = {
				$ACTION$_character_modifier = flag_kkt_4_modifier
			}
			flag:flag_kkt_5 = {
				$ACTION$_character_modifier = flag_kkt_5_modifier
			}
			flag:flag_kkt_6 = {
				$ACTION$_character_modifier = flag_kkt_6_modifier
			}
		}
	}
	hidden_effect = { # 实际执行
		if = {
			limit = {
				$CONFIRM$ = yes #发布国策
			}
			switch = {
				trigger = $FLAG$
		
				flag:flag_kkt_1 = {
					$ACTION$_character_modifier = flag_kkt_1_modifier
				}
				flag:flag_kkt_2 = {
					$ACTION$_character_modifier = flag_kkt_2_modifier
				}
				flag:flag_kkt_3 = {
					$ACTION$_character_modifier = flag_kkt_3_modifier
				}
				flag:flag_kkt_4 = {
					$ACTION$_character_modifier = flag_kkt_4_modifier
				}
				flag:flag_kkt_5 = {
					$ACTION$_character_modifier = flag_kkt_5_modifier
				}
				flag:flag_kkt_6 = {
					$ACTION$_character_modifier = flag_kkt_6_modifier
				}
			}
		}
		if = {
			limit = {
				$CONFIRM$ = no #撤销国策
			}
		}
	}
}

SPT_kkt_set_cost_to_character = { #国策花费 只会在发布时候应用
	switch = {
		trigger = $FLAG$

		flag:flag_kkt_1 = {
			remove_short_term_gold = 10
		}
		flag:flag_kkt_2 = {
			remove_short_term_gold = 20
		}
		flag:flag_kkt_3 = {
			remove_short_term_gold = 30
		}
		flag:flag_kkt_4 = {
			remove_short_term_gold = 40
		}
		flag:flag_kkt_5 = {
			remove_short_term_gold = 50
		}
		flag:flag_kkt_6 = {
			remove_short_term_gold = 60
		}
	}
}
