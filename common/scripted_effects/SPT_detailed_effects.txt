SPT_courtly_vassal_towards_policy_opinion_effect = {
	if = {
		limit = { any_vassal = { has_vassal_stance = courtly } }
		save_scope_as = actor
		if = {
			limit = { $POLICY$ = neutral }
			custom_tooltip = SPT_courtly_vassal_towards_policy_opinion_effect_tt
			#hidden_effect = {
				every_vassal = {
					limit = {
						has_vassal_stance = courtly
					}
					add_opinion = {
						target = scope:actor
						modifier = spt_vassal_towards_neutral_policy_opinion
						opinion = $OPINION$
					}
				}
			#}
		}
	}
}
