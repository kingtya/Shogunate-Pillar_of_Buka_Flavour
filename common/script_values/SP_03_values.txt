			limit = {

		}

			}
			add = 7.5
		}
		if = {
			limit = {
				scope:target = {
					this.house.house_head = scope:owner
				}
			}
			add = 15
		}
		if = { 
			limit = {
				scope:target = {
					NOT = { is_close_family_of = scope:owner }
					is_extended_family_of = scope:owner
				}
			}
			add = 7.5
		}
		if = { 
			limit = {
