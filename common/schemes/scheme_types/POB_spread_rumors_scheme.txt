pob_spread_rumors = {
	# Basic Setup
	skill = intrigue
	desc = pob_spread_rumors_desc_general
	success_desc = "BEFRIEND_SUCCESS_DESC"

	icon = intrigue
	illustration = "gfx/interface/illustrations/event_scenes/corridor.dds"
	category = hostile
	target_type = character
	is_secret = yes
	is_basic = yes
	maximum_breaches = 2
	cooldown = {
		years = 3
	}

	# Parameters
	speed_per_skill_point = -2.5
	spymaster_speed_per_skill_point = 0	
	uses_resistance = yes
	base_progress_goal = 170
	maximum_secrecy = 85
	base_maximum_success = 95
	minimum_success = 5
	
	# Core Triggers
	allow = {
		is_adult = yes
		is_imprisoned = no
		scope:target = {
			is_adult = yes
			is_imprisoned = no
			is_ruler = yes
		}
		pob_has_any_running_task_scheme = no
	}

	valid = {
		is_imprisoned = no
		is_incapable = no
		scope:target = {
			is_ruler = yes
			is_imprisoned = no
			is_incapable = no
			exists = location
			in_diplomatic_range = scope:owner
		}
	}

	odds_prediction = {
		add = base_odds_prediction_target_is_char_value
		add = odds_skill_contribution_intrigue_value
		min = 0
	}

	pulse_actions = {
		entries = {
		}
		chance_of_no_event = scheme_apa_chance_of_no_event_value
	}

	base_secrecy = {
		add = secrecy_base_value
		add = countermeasure_apply_secrecy_maluses_value
	}

	# Base Chances
	base_success_chance = {
		base = 25
		
		modifier = {
			add = {
				value = scope:owner.intrigue
			 	multiply = 2
			}
			desc = SCHEME_YOUR_INTRIGUE
		}

		modifier = {
			add = 20
			scope:owner = {
				has_trait = ninja
			}
			desc = POB_SCHEME_OWNER_NINJA
		}

		apply_opportunistic_scheme_success_chance_adjustments_modifier = yes

		# 家族团结度
		modifier = {
 			add = {
				value = scope:owner.house.house_unity_value
				multiply = 0.25
			}
 			desc = SP_AI_HOUSE_UNITY
			exists = scope:owner.house
			scope:target = {
				exists = house
				house = scope:owner.house
				house = {
					OR = {
						has_house_unity_stage = harmonious
						has_house_unity_stage = friendly
					}
				}
			}
 		}

	}
	
	# On Actions
	on_start = {
		set_variable = {
			name = apply_countermeasures
			value = flag:opportunistic
		}
	}

	on_phase_completed = {
		save_scope_as = scheme
		scheme_target_character = { save_scope_as = target }
		scheme_owner = { save_scope_as = owner }
		scope:scheme.scheme_owner = {
			trigger_event = pob_spread_rumors_events.0002
		}
	}

	on_invalidated = {
		scheme_target_character = {
			save_scope_as = target
		}
		scheme_owner = {
			save_scope_as = owner
		}

		if = {
			limit = {
				scope:target = { 
					NOT = { in_diplomatic_range = scope:owner } 
				}
			}
			scope:owner = {
				send_interface_toast = {
					title = pob_spread_rumors_invalidated_title
					left_icon = scope:target
					custom_description_no_bullet = {
						object = scope:target
						text = scheme_target_not_in_diplomatic_range
					}
				}
			}
		}
	}

	on_monthly = {
		hostile_scheme_monthly_discovery_chance_effect = yes
	}
}
